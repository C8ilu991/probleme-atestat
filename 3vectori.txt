#include <iostream>
using namespace std;

int main(){
    int n,x;
    cin >> n;
    x = n%10;
    n /= 10;
    while(x <= n%10){
        x = n%10;
        n /= 10;
    }
    while(x >= n%10 and n!=0){
        x = n%10;
        n /= 10;
    }
    if(n > 0)
        cout << "NU";
    else
        cout << "DA";
    return 0;
}
