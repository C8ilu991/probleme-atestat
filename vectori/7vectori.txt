#include <iostream>
#include <fstream>
using namespace std;

ifstream fin("cifre.txt");

int main(){
    int n, nr;
    fin >> n;
    for(int i=1;i<=n;++i){
        fin >> nr;
        if(nr >= 100 and nr <= 900)
            if(nr%10 != (nr/10)%10 and (nr/10)%10 != (nr/100)%10 and nr%10!=(nr/100)%10)
                cout << nr << " ";
    }

    return 0;   
}
